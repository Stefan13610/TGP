\subsection{ODE solitonu przy zmiennym sprzńôŇľeniu kinetycznym}
\label{app:L-ode}

\subsubsection{R√≥wnanie bazowe}

ODE solitonu sferycznie symetrycznego (por.\ dodatek~\ref{app:ogon-masy}):
\begin{equation}\label{eq:L-ode}
  f_\alpha(g)\,g'' + \frac{2}{r}\,g' = V'(g),
  \qquad
  f_\alpha(g) = 1 + 2\alpha_{\rm kin}\ln g,
  \qquad
  V'(g) = g^2(1-g),
\end{equation}
z~warunkami: $g(0) = g_0$, $g'(0) = 0$, $g(r)\to 1$ dla $r\to\infty$.

\textbf{Uwaga fizyczna:} dla $\alpha_{\rm kin} = 2$ (wartoŇõńá TGP)
otrzymujemy identyczne ODE jak w~dodatkach~J--K. Skan po $\alpha_{\rm kin}$
bada \emph{rodzinńô} ODE sparametryzowanńÖ tym wsp√≥Ňāczynnikiem.

\subsubsection{Punkt osobliwy i~bariera kinetyczna}

Czynnik $f_\alpha$ zeruje sińô przy:
\begin{equation}\label{eq:L-gstar}
  g^*(\alpha_{\rm kin}) = \exp\!\left(-\frac{1}{2\alpha_{\rm kin}}\right),
  \qquad
  g^*(2) = e^{-1/4} \approx 0{,}779.
\end{equation}
Bariera kinetyczna ($g \geq g^*$) obowińÖzuje dla kaŇľdego $\alpha_{\rm kin} > 0$.

\subsubsection{Punkt strzaŇāu $z_0(\alpha_{\rm kin})$: czysta sinusowa faza}

SpoŇõr√≥d cińÖgŇāego spektrum $g_0 \in (g^*, \infty)$ wyr√≥Ňľniamy punkt
\textbf{czystej fazy sinusowej}:
\begin{equation}\label{eq:L-z0-def}
  z_0(\alpha_{\rm kin})\colon\qquad B_{\rm tail}(z_0, \alpha_{\rm kin}) = 0,
\end{equation}
gdzie $B_{\rm tail}$ jest amplitudńÖ kosinusowńÖ ogona (por.\ dodatek~J,
hipoteza H1, stwierdzenie~\ref{prop:J-quantization}).
$z_0$ jest pierwszym zerem $B_{\rm tail}$ w~sektorze $n_{\rm bounce}=0$.

\begin{remark}[WartoŇõci numeryczne]
Dla fizycznego $\alpha_{\rm kin} = 2$:
$z_0(2) = g_0^e \approx 1{,}230$ (elektron, wyniki \texttt{ex58}, 0{,}8\% od $1{,}24$).
\end{remark}

\begin{hypothesis}[FormuŇāa sumy krytycznych sprzńôŇľeŇĄ]%
\label{hyp:L-sum-formula}
Suma $\alpha^*_1 + \alpha^*_2$ jest dokŇāadnie wyraŇľona przez
substratowe staŇāe TGP:
\begin{equation}\label{eq:L-sum-exact}
  \alpha^*_1 + \alpha^*_2
  \;=\; 2\pi - \frac{\alpha_{\rm TGP}}{2} - \frac{\beta_{\rm pot}}{10}
  \;=\; 2\pi - 1 - \frac{1}{10}
  \;=\; 2\pi - \frac{11}{10},
\end{equation}
gdzie:
\begin{itemize}[nosep]
  \item $\alpha_{\rm TGP} = 2$ (substratowy wykŇāadnik kinetyczny, N0-2),
  \item $\beta_{\rm pot} = 1$ (normalizacja potencjaŇāu $V'(g) = \beta_{\rm pot}\,g^2(1-g)$, N0-5).
\end{itemize}

\textbf{Motywacja}: skrypt \texttt{ex73} weryfikowaŇā og√≥lnoŇõńá formuŇāy
$S = 2\pi - 1 - \beta_{\rm pot}/10$ dla r√≥Ňľnych wartoŇõci $\beta_{\rm pot}$.

\textbf{prawem strukturalnym} --- nie numerologińÖ dla $\beta_{\rm pot} = 1$.

Status: \statuslabel{Hipoteza} \textbf{obalona} (v36, O-L1 zamkni\k{e}te):
$S(\infty) = 5{,}0674 \neq 2\pi{-}11/10 = 5{,}1832$ ($22336\,\text{ppm}$,
\texttt{ex88}).
Brak prostej formu\l{}y algebraicznej dla~$S$.
Otwarte: (ii)~weryfikacja og√≥lno\'sci przez skan $\beta_{\rm pot}$
(\texttt{ex89}, O-L2), (iii)~analityczna derywacja $\alpha^*_{1,2}$.
\end{hypothesis}

\begin{remark}[ZwińÖzek z~$\alpha_{\rm TGP}$ i~$2\pi$]
FormuŇāa~\eqref{eq:L-sum-exact} zawiera $2\pi$ --- peŇāny cykl fazy
w~opisie solitonu oscylacyjnym. CzŇāon $\alpha_{\rm TGP}/2 = 1$ jest
\emph{poŇāowńÖ} substratowego wykŇāadnika kinetycznego; czŇāon $1/10$ pochodzi

tak, Ňľe r√≥Ňľnica ich odlegŇāoŇõci od $\pi$ (centrum) jest wyznaczona przez

Jest to \textbf{warunek p√≥Ňāsumy}: struktura spektrum $F(\alpha)$ jest
symetrycznie eksponowana wok√≥Ňā $\pi - (11/10)/2 = \pi - 11/20$.
\end{remark}
