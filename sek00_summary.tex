% ----------------------------------------------------------
\subsection*{Aksjomaty TGP (A1--A8)}

\begin{enumerate}[label=\textbf{A\arabic*}.,leftmargin=2em]
\item \textbf{Przestrze\'n z~materii.}
  Przestrze\'n nie istnieje niezale\.znie; jest generowana
  przez obecno\'s\'c masy/energii (\S\ref{sec:ontologia}).
\item \textbf{Nico\'s\'c $\Nzero$.}
  Stan bez materii i~pola: $\Phi = 0$, brak przestrzeni (\S\ref{ssec:N0}).
\item \textbf{Zasada zerowej sumy.}
  Zamkni\k{e}ty Wszech\'swiat w~ca\l{}o\'sci: $\sum \Phi = 0$ (\S\ref{sec:ontologia}).
\item \textbf{Niestabilno\'s\'c $\Nzero$.}
  Stan $\Phi = 0$ jest niestabilny --- Wielki Wybuch jako
  przej\'scie fazowe (\S\ref{subsec:N0_instability}, dod.~\ref{app:wielki-wybuch}).
\item \textbf{R\'ownanie pola.}
  Ka\.zda cz\k{a}stka wytwarza lokalny wk\l{}ad:
  $\mathcal{D}[\Phi_a] = q_a \cdot \rho_a$ (\S\ref{sec:pole}).
\item \textbf{Pr\k{e}dko\'s\'c \'swiat\l{}a.}
  $c(\Phi) = c_0 (\PhiZero/\Phi)^{1/2}$ (\S\ref{sec:stale}).
\item \textbf{Emergentna sta\l{}a Plancka.}
  $\hbar(\Phi) = \hbar_0 (\PhiZero/\Phi)^{1/2}$ (\S\ref{sec:stale}).
\item \textbf{Sta\l{}a grawitacyjna.}
  $G(\Phi) = G_0 \cdot \PhiZero/\Phi$ (\S\ref{sec:stale}).
\end{enumerate}
\noindent
Wyk\l{}adniki $(a, b, g) = (1/2,\, 1/2,\, 1)$ s\k{a}
\emph{jedynym} rozwi\k{a}zaniem
zachowuj\k{a}cym sta\l{}o\'s\'c d\l{}ugo\'sci Plancka
$\ell_P = \sqrt{\hbar G/c^3} = \mathrm{const}$
(tw.~\ref{thm:exponents}).

\subsection*{Dzia\l{}anie zunifikowane}

Dzia\l{}anie TGP (\S\ref{ssec:unified-action}, hipoteza~\ref{hyp:unified-action}):
\begin{equation}\label{eq:S-TGP-summary}
S_{\mathrm{TGP}}[\Phi,\psi_m]
= \int d^4x\;\sqrt{-g_{\mathrm{eff}}}\;\Bigl[
  \underbrace{\tfrac{1}{2}\,K(\varphi)\,
  (\partial\varphi)^2 - V(\varphi)}_{\text{sektor pola}}
  \;-\;
  \underbrace{\tfrac{q}{\PhiZero}\,\varphi\,\rho}_{\text{sprz\k{e}\.zenie z~materi\k{a}}}
\Bigr]
\end{equation}
gdzie $\varphi = \Phi/\PhiZero$,\;
$K(\varphi) = K_{\mathrm{geo}}\,\varphi^4$,\;
$V(\varphi) = \frac{\gamma}{3}\varphi^3 - \frac{\gamma}{4}\varphi^4$
(warunek $\beta = \gamma$, tw.~\ref{thm:beta_gamma}).
Metryka efektywna:
\begin{equation}
ds^2 = -\frac{c_0^2}{\varphi}\,dt^2 + \varphi\,\delta_{ij}\,dx^i dx^j
\end{equation}

\subsection*{G\l{}\'owne r\'ownania pola}

\noindent\textbf{R\'ownanie stacjonarne} (soliton, $d=3$):
\begin{equation}
g^2 g'' + g(g')^2 + \frac{2}{r}\,g^2 g' = g^2(1 - g)
\end{equation}
z~warunkiem: $g(0) = g_0$, $g'(0) = 0$, $g(\infty) \to 1$.

\noindent\textbf{Granica s\l{}abego pola} (PPN, uwaga~\ref{rem:PPN-Will}):
$\gamma_{\mathrm{PPN}} = 1$,\;
$\beta_{\mathrm{PPN}} = 1$
--- zgodne z~GR.

\noindent\textbf{Fale grawitacyjne} (\S\ref{ssec:tensor-substrate}, dod.~\ref{app:ringdown}):
$c_{\mathrm{GW}} = c_0$ dok\l{}adnie na tle kosmologicznym
(stw.~\ref{prop:cT}); zgodne z~GW170817.

\noindent\textbf{Kosmologia} (FLRW, \S\ref{sssec:cosmo-complete}):
\begin{equation}
H^2(a,\varphi) = \frac{1}{\varphi(a)}\,H^2_{\Lambda\mathrm{CDM}}(a),
\qquad
\kappa = \frac{3}{4\PhiZero} \approx 0{,}030
\end{equation}

\subsection*{Sektor cz\k{a}steczkowy --- predykcje ilo\'sciowe}

TGP przewiduje masy lepton\'ow i~sta\l{}\k{a} sprz\k{e}\.zenia
silnego z~\textbf{jednego parametru substratowego} $g_0^e = 0{,}869$,
wyznaczonego przez ODE solitonu i~z\l{}ot\k{a} proporcj\k{e}
$\varphi = (1+\sqrt{5})/2$:

\begin{itemize}[nosep]
\item \textbf{Mechanizm $\varphi$-FP}: $g_0^\mu = \varphi \cdot g_0^e$
  \;\;$\Rightarrow$\;\; $r_{21} = [A_{\rm tail}(\varphi g_0)/A_{\rm tail}(g_0)]^4 = 206{,}77$
  (PDG: $206{,}768$; odch.\ $0{,}001\%$).
\item \textbf{Koide} ($K = 2/3$): zamyka $m_\tau$ --- siedem predykcji mas z~precyzj\k{a} $< 0{,}1\%$.
\item \textbf{Stala sprz\k{e}\.zenia silnego}:
  $\alpha_s(M_Z) = N_c^3\, g_0^e/(8\,N_f^2) = 0{,}1174$
  (PDG: $0{,}1179 \pm 0{,}0009$; $0{,}6\sigma$).
\item \textbf{Discrete running}:
  $\alpha_s(m_\tau) = 3\,g_0^e/8 = 0{,}326$
  (PDG: $0{,}330 \pm 0{,}014$; $0{,}3\sigma$).
\item \textbf{Ratio (zero parametr\'ow)}:
  $\alpha_s(m_\tau)/\alpha_s(M_Z) = (5/3)^2 = 25/9 = 2{,}778$
  (PDG: $2{,}799 \pm 0{,}121$; $0{,}18\sigma$).
\end{itemize}
