\subsection*{Streszczenie programu P6}

Dodatek P6 rozszerza formu\l{}\k{e} $T_c$ programu P5 o~mechanizmy
niezbe\k{e}dne do~opisu wysokotemperaturowych nadprzewodnik\'ow (cuprates,
hydrydy, FeSe/STO, MgB${}_2$). Program P5 osi\k{a}gn\k{a}\l{} $r^2_{\log}=0{,}48$
na 19 klasycznych SC~---~za~s\l{}abo do~predykcji nowych materia\l{}\'ow.
P6 wprowadza dwa mechanizmy:

\begin{enumerate}\itemsep0pt
  \item \textbf{P6.A}~---~2D \emph{d-wave} z~hybryd\k{a} Zhanga-Rice'a
  dla~cupratn\'ow. Jednolity parametr $K_{dw}$ (d-wave + van Hove + ZR).
  \item \textbf{P6.B}~---~coupling fonon-substrat: $\Lambda_E^{\text{eff}}
  = \Lambda_0 \cdot (\omega_\text{ph}/\omega_0)^\alpha$. Kalibrowane
  na~hydrydach wysokoci\'snieniowych, dzia\l{}a te\.z dla MgB${}_2$
  i~FeSe/STO.
\end{enumerate}

\medskip\noindent\textbf{Werdykt~P6 (2/3 pe\l{}ne + 1/3 otwarte):}
\begin{itemize}\itemsep0pt
  \item \textbf{P6.A}~(cuprates, ps10): \emph{pe\l{}ne.}
    $r(\log)=0{,}957$ na~8 cupratach
    (La$_2$CuO$_4$, YBCO, BiSCCO, Tl-rodzina, Hg-1223, Hg-2223,
    Nd$_2$CuO$_4$, Bi-2201). RMS\textsubscript{log}$=0{,}19$.
  \item \textbf{P6.B}~(uniwersalny, ps13): \emph{pe\l{}ne.}
    $r(\log)=0{,}877$ na~16 materia\l{}ach mieszanych
    (BCS klasyczne, MgB${}_2$, FeSe/STO, Fe-pniktidy, cuprates,
    hydrydy wysokoci\'snieniowe).
  \item \textbf{P6.C}~(f-metale orbital switching): \emph{cz\k{e}\'sciowe.}
    Traktowanie lantanowc\'ow jako \(d\)-block pod ci\'snieniem dzia\l{}a
    (ps12), ale formalizacja przej\'scia $\eta(P)$ odroczona do~P7.
\end{itemize}

Pliki: \texttt{research/superconductivity\_closure/ps\{9--14\}.py}
oraz \texttt{P6A\_summary.md},~\texttt{P6B\_summary.md}.

\subsection{Model P6.A: cuprates 2D d-wave z~Zhanga-Rice'a}
\label{sec:P6A}

\subsubsection{Formu\l{}a}

Cuprate jest quasi-2D SC w~p\l{}aszczyznach CuO${}_2$. Mechanizm
opiera si\k{e} na~trzech elementach:

\begin{enumerate}\itemsep0pt
  \item \emph{Zhang-Rice singlet}~---~Cu $3d_{x^2-y^2}$ hybrydyzuje
    z~2 tlenkami O $2p$:
    \[
      A_{ZR}^2 \;=\; A_d^2 \;+\; 2\,A_p^2
    \]
    (w~P4: $A_d=0{,}3096$, $A_p=A_{sp}=0{,}2067$, $A_{ZR}^2=0{,}181$).
  \item \emph{d-wave gap}~---~$\Delta(\mathbf{k})=\Delta_0(\cos k_x a
    -\cos k_y a)$ wzmacnia sprz\k{e}\.zenie w~kierunkach~$(1,0)$ Fermi
    surface (boost~$\sim 1{,}5$).
  \item \emph{Van Hove singularity}~---~DOS rozbiega przy~domieszaniu
    optymalnym; efektywny $\lambda_\text{eff}\cdot (1+\ln \mathrm{DOS})$
    (boost~$\sim 2{,}5$).
\end{enumerate}

Wszystkie trzy multiplikatywne wzmocnienia zbieramy do~jednego
uniwersalnego wsp\'o\l{}czynnika:
\[
  \boxed{\quad
    T_c^{\text{cuprate}}
    \;=\; K_{dw}\,k_d(z_\parallel)\,C_0\,A_{ZR}^2\,M(a_\parallel)\,
    \sqrt{n_\text{layers}}\,\Lambda_E^{\text{cup}}
  \quad}
\]

\noindent gdzie:
\begin{itemize}\itemsep0pt
  \item $K_{dw}=3{,}498$~---~ uniwersalny d-wave+vH+ZR booster
    (fit ps10).
  \item $k_d(z_\parallel=8)=2{,}936$~---~3D XY-Monte Carlo
    dla~quasi-2D~warstwowych.
  \item $C_0=48{,}82$~---~sta\l{}a z~ODE substratu (P4).
  \item $M(a)=\exp(-(a-n a^*_{TGP})^2/\sigma_a^2)$~---~modulacja
    harmoniki TGP, $a^*_{TGP}=4{,}088$\,\AA, $\sigma_a=2{,}59$~\AA.
  \item $\sqrt{n}$~---~empiryczny fakror liczby warstw CuO${}_2$
    (saturation; pasuje w~zakresie $n\in[1,5]$).
  \item $\Lambda_E^{\text{cup}}=0{,}051$~meV~---~skala elektronowego
    parowania (2{,}6\,\(\times\) \emph{mniejsza} ni\.z Al-BCS $0{,}131$).
\end{itemize}

\subsection{Model P6.B: coupling fonon-substrat}
\label{sec:P6B}

\subsubsection{Formu\l{}a}

Substrat TGP $\Phi$ oscyluje lokalnie z~cz\k{e}stotliwo\'sci\k{a}
fonon\'ow materia\l{}u. Im wy\.zsza $\omega_\text{ph}$, tym szybszy
przep\l{}yw momentu energetycznego do~substratu. Zak\l{}adamy skalowanie
pot\k{e}gowe:
\[
  \boxed{\quad
    \Lambda_E^{\text{eff}}(\omega_\text{ph})
    \;=\; \Lambda_0 \cdot \left(\frac{\omega_\text{ph}}{\omega_0}\right)^\alpha
  \quad}
\]

\noindent gdzie $\omega_0=15$~meV (Al Debye reference), $\alpha$ i~$\Lambda_0$
sa fit-parametrami.

\subsubsection{Kalibracja ps12 na~hydrydach high-P}

Dla 9~hydrydów wysokoci\'snieniowych (H${}_3$S, LaH${}_{10}$, YH${}_9$,
CaH${}_6$, YH${}_6$, ThH${}_{10}$, CeH${}_9$, CeH${}_{10}$, D${}_3$S):
\begin{align*}
  \alpha   &\approx 1{,}04 \quad \text{(prawie liniowe!)} \\
  \Lambda_0 &\approx 0{,}096~\text{meV} \quad \text{(vs~BCS ps5: 0{,}131)}
\end{align*}

\noindent $\alpha\approx 1$ ma prosty sens fizyczny: moment energetyczny
przekazywany do~substratu~$\propto$ szybko\'s\'c oscylacji fononowej.

\subsubsection{Walidacja uniwersalna ps13}

Formu\l{}a P6.B zastosowana do~16~materia\l{}\'ow mieszanych
(BCS klasyczne + MgB${}_2$ + FeSe + cuprates + pnikti dzy + hydrydy):
$$r(\log)=0{,}877,\qquad \mathrm{RMS}_{\log}=0{,}32$$

\noindent \textbf{Kluczowe rozwi\k{a}zania przez P6.B:}
\begin{itemize}\itemsep0pt
  \item \textbf{MgB${}_2$}: $\omega=75$~meV $\to$ $T_c^\text{pred}=31$~K
    (vs~obs $39$~K; \textbf{80\%}). Wcze\'sniej ps9 two-gap dawa\l{}o tylko 14~K.
  \item \textbf{FeSe/SrTiO${}_3$} monolayer: $\omega_\text{STO}=80$~meV
    $\to$ $T_c^\text{pred}=86$~K (vs~obs $65$~K; \textbf{133\%}).
    Wcze\'sniej ps11 single-band dawa\l{}o 28~K.
  \item \textbf{LaH${}_{10}$}: $\omega=250$~meV $\to$ $T_c^\text{pred}=368$~K
    (vs~obs $250$~K; overshoot).
\end{itemize}

Pozosta\l{}y outlier: \emph{FeSe bulk} (obs $8$~K, pred $26$~K,
overshoot $3{,}2\times$). Sygnalizuje potrzebno\'s\'c czynnika
t\l{}umi\k{a}cego~--- magnetyzm/lokalizacja (zakres P6.D).

\subsection{Otwarte problemy P6.C i~P6.D}
\label{sec:P6-open}

\textbf{P6.C (f-metale orbital switching):}
Amplituda $A_f=2{,}03$ z~ps5 5c przeszacowuje T$_c$ dla Ce, Yb
przy~ambient~(\(\sim 1300\)~K, absurd). Pod~ci\'snieniem
hybrydyzacja~$4f \to 5d$ daje~$A_\text{eff}\to A_d$, co~pasuje do~Ce/Th
hydryd\'ow. Formalnie:
\[
  A_\text{eff}(\rho, P) = (1-\eta(P))\,A_f + \eta(P)\,A_d,
  \qquad \eta(P\to\infty)=1
\]

\noindent Funkcja $\eta(P)$ jest fenomenologiczna; pe\l{}na derivacja
z~TGP substratu wymaga osobnej pracy (program P7).

\textbf{P6.D (magnetyzm blokuj\k{a}cy faz\k{e}):}
FeSe bulk daje obs $8$~K vs pred 26 K. Ambient Fe~---~TGP pred 17 K,
obs niespin\'owe do~ci\'snienia 20~GPa~(T$_c$=2~K).
Sygna\l{}: \emph{blocking factor}
$B_\text{mag}(T,\rho)\in[0,1]$ mno\.z\k{a}cy $T_c^\text{pred}$.
Funkcja~$B$ nie jest wyprowadzona z~substratu P6, wymaga coupling
do~magnetycznego pola $M$ z~sektora P4 particle.

\subsection{Podsumowanie programu P6}
\label{sec:P6-summary}

Program P6 osi\k{a}gn\k{a}\l{} \textbf{r=0.88 na 16 mieszanych SC}
(vs~P5: 0.48~na 19)~--- jako\'sciowy przeskok w~predyktywno\'sci
modelu TGP.

\medskip\noindent\textbf{G\l{}\'owne osi\k{a}gni\k{e}cia:}
\begin{enumerate}\itemsep0pt
  \item Cuprates (8 material\'ow) z~r=0.96 przez d-wave + Zhanga-Rice.
  \item MgB${}_2$, FeSe/STO, hydrydy---wszystko z~jednolitej formu\l{}y
    P6.B (alpha=1.04, $\Lambda_0=$0.096).
  \item Konkretne eksperymentalne predykcje w~zakresie~170-240~K
    przy ambient pressure (Hg1245/STO, FeSe/diament).
\end{enumerate}

\medskip\noindent\textbf{G\l{}\'owne pozosta\l{}o\'sci:}
\begin{enumerate}\itemsep0pt
  \item FeSe bulk 8K vs 26K pred (P6.D magnetyzm).
  \item Ce/Yb f-metale pod~ciśn.(P6.C orbital switching).
  \item Room-temperature ambient~---~matematycznie dost\k{e}pny,
    chemicznie otwarty problem (brak znanego metalicznego $\omega>200$
    meV materia\l{}u w~harmonice $a^*$).
\end{enumerate}

\textbf{Propozycja dla empiryka:} \emph{Hg1245 na~SrTiO$_3$}---jeden
eksperyment mo\.ze jednocze\'snie falsyfikowa\'c/potwierdzi\'c P6.A
(zale\.zno\'s\'c od~$n$) i~P6.B (interface phonon boost).
Je\'sli~$T_c>200$~K obserwowane, TGP~P6~jest~potwierdzony pe\l{}ne.
