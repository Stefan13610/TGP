\subsection{Sektor cz\k{a}steczkowy --- najsilniejszy argument TGP}%

\begin{remark}[Dlaczego sektor cz\k{a}steczkowy jest kluczowy]%
\label{rem:particle-sector-highlight}
TGP \textbf{nie jest tylko teori\k{a} grawitacji}: jako teoria skalarnego
substratu, z~kt\'orego wynika metryka, sta\l{}e fundamentalne
\emph{i}~masy cz\k{a}stek, TGP posiada sektor cz\k{a}steczkowy
daj\k{a}cy ilo\'sciowe predykcje, jakich nie potrafi dor\'owna\'c
\.zadna inna modyfikacja grawitacji.

\medskip\noindent
\textbf{1.\ Masy lepton\'ow z~jednego parametru ($g_0^e$).}
ODE substratowe $g^2g'' + g(g')^2 + \frac{2}{r}g^2g' = g^2(1-g)$
posiada 1-parametrow\k{a} rodzin\k{e} rozwi\k{a}za\'n;
parametr centralny $g_0^e = 0{,}869$ jest kalibrowany
przez stosunek mas $r_{21} = m_\mu/m_e$
(mechanizm $\varphi$-FP, dod.~J2).
Z~jednego $g_0^e$ wynikaj\k{a}:
\begin{itemize}[nosep]
  \item $r_{21} = [A_{\rm tail}(\varphi g_0^e)/A_{\rm tail}(g_0^e)]^4
        = 206{,}77$ \quad(PDG: $206{,}768$; $0{,}001\%$),
  \item $m_\tau$ przez relacj\k{e} Koide ($K = 2/3$): $1776{,}9$~MeV
        \quad(PDG: $1776{,}86$; $0{,}002\%$),
  \item siedem stosunk\'ow mas lepton\'ow z~precyzj\k{a} ${<}\,0{,}1\%$
        (tab.~\ref{res:pred-summary-v45}).
\end{itemize}

\medskip\noindent
\textbf{2.\ Sta\l{}a sprz\k{e}\.zenia silnego --- zero wolnych parametr\'ow.}
Ten sam $g_0^e$ (wyznaczony wy\l{}\k{a}cznie z~mas lepton\'ow)
daje formu\l{}\k{e} (dod.~V, rem.~\ref{rem:V-alphas-alpha1}):
\begin{equation}\label{eq:alphas-highlight}
  \alpha_s(M_Z) = \frac{N_c^3\,g_0^e}{8\,N_f^2}
  = \frac{27 \times 0{,}869}{8 \times 25} = 0{,}1174
  \qquad (\text{PDG: } 0{,}1179 \pm 0{,}0009;\; 0{,}6\sigma).
\end{equation}
Jest to predykcja z~\textbf{zerowym} wolnym parametrem kosmologicznym
czy cz\k{a}steczkowym --- $g_0^e$ pochodzi z~sektora lepton\'ow.

\begin{quote}
\textbf{Nota kanoniczna.} Formu\l{}a kanoniczna na sta\l{}\k{a} sprz\k{e}\.zenia silnego to
$\alpha_s(M_Z) = 3g_0^e/(32\Omega_\Lambda) = 0{,}1190$ (F1),
z~$g_0^e = 0{,}86941$.
Powy\.zsza formu\l{}a $\alpha_s = N_c^3 g_0^e/(8N_f^2)$ jest z~ni\k{a} zgodna
do $1{,}4\%$ dzi\k{e}ki relacji
$\Phi_0 \approx N_f^2 \approx 36\Omega_\Lambda \approx 25$.
Patrz: \texttt{tgp\_companion.tex}, Sec.~3.1.
\end{quote}

\medskip\noindent
\textbf{3.\ Discrete running --- test bezparametrowy.}
Stosunek $\alpha_s$ przy dw\'och progach smak\'ow
($N_f = 3$ przy $m_\tau$, $N_f = 5$ przy $M_Z$)
jest predykcj\k{a} \emph{bez jakichkolwiek parametr\'ow}:
\begin{equation}\label{eq:ratio-highlight}
  \frac{\alpha_s(m_\tau)}{\alpha_s(M_Z)}
  = \left(\frac{5}{3}\right)^{\!2} = \frac{25}{9} = 2{,}778
  \qquad (\text{PDG: } 2{,}799 \pm 0{,}121;\; 0{,}18\sigma).
\end{equation}

\medskip\noindent
\textbf{Implikacja.}
Ca\l{}y \l{}a\'ncuch $r_{21} \to g_0^e \to \alpha_s(M_Z)$
tworzy zamkni\k{e}t\k{a} przeobraz\k{e}: z~jednego obserwabla
elektros\l{}abego (stosunek mas lepton\'ow) wynika obserwabla
QCD (sta\l{}a sprz\k{e}\.zenia silnego). Jest to cecha unikalna TGP,
niemo\.zliwa w~teoriach czysto grawitacyjnych.

Weryfikacja numeryczna: skrypty \texttt{ex186} (kalibracja $g_0^e$),
\texttt{ex184} (discrete running), \texttt{ex185} (analiza running).
\end{remark}

\subsection{Argument za wymiarem $3{+}1$}\label{ssec:wymiar}

Pytanie ,,dlaczego trzy wymiary przestrzenne?'' w~TGP staje
si\k{e} pytaniem o~w\l{}asno\'sci substratu relacyjnego $\Gamma$
i~przej\'scia fazowego $\Ffield = 0 \to \Ffield > 0$.

\begin{proposition}[Wymiar $d = 3$ jako optymalny
  dla TGP]\label{prop:wymiar}
Substrat relacyjny $\Gamma$ o~efektywnym wymiarze grafowym~$d$
generuje stabiln\k{a} faz\k{e} metryczn\k{a} ($\Phi > 0$)
z~bogat\k{a} struktur\k{a} defektow\k{a} \textbf{jedynie
dla $d = 3$}. Argument opiera si\k{e} na analogii z~modelem
Isinga na grafach:
\begin{enumerate}[nosep]
  \item \textbf{$d = 1$}: Brak przej\'scia fazowego
        (twierdzenie Mermina--Wagnera w~$1D$).
        Substrat nie mo\.ze wygenerowa\'c stabilnej metryki.
  \item \textbf{$d = 2$}: Przej\'scie fazowe istnieje
        (Ising~$2D$), ale defekty (\'scianki domenowe)
        maj\k{a} wymiar $d{-}1 = 1$ --- s\k{a} niestabilne,
        rozpadaj\k{a} si\k{e}. Faza metryczna jest krucha;
        defekty nie mog\k{a} tworzy\'c stabilnych cz\k{a}stek.
  \item \textbf{$d = 3$}: Przej\'scie fazowe stabilne.
        Defekty tworz\k{a} bogat\k{a} hierarchi\k{e}:
        \'scianki ($2D$) s\k{a} stabilne, linie ($1D$) i~punkty
        ($0D$) r\'ownie\.z --- daj\k{a}c naturaln\k{a}
        podstaw\k{e} dla r\'o\.znorodnej materii
        (sekcja~\ref{ssec:defekty}).
  \item \textbf{$d \geq 4$}: Przej\'scie fazowe staje si\k{e}
        trywialne (przybli\.zenie pola \'sredniego jest
        dok\l{}adne). Fluktuacje s\k{a} nieistotne ---
        brak bogatej struktury defektowej, brak hierarchii mas.
\end{enumerate}
\end{proposition}

\begin{remark}[Wymiar czasowy]
Wymiar $+1$ (czas) wynika z~definicji czasu w~TGP
(sekcja~\ref{ssec:czas}) jako parametru porz\k{a}dkuj\k{a}cego
stany w~$\mathcal{S}_1$. Porz\k{a}dek jest \textbf{liniowy}
(relacja przyczynowa jest totalnym porz\k{a}dkiem na \'scie\.zkach),
co daje dok\l{}adnie jeden wymiar czasowy.
Argument ten jest \textbf{preferencyjny}, nie dedukcyjny ---
nie dowodzi, \.ze $d = 3$ jest jedynym mo\.zliwym wymiarem,
lecz \.ze jest jedynym, w~kt\'orym TGP produkuje fizyk\k{e}
o~obserwowanej z\l{}o\.zono\'sci.
\end{remark}
