\subsection{Formu\l{}a Koide'go w~TGP}%
\label{app:T-koide-def}
% -------------------------------------------------------------------

\begin{definition}[Relacja Koide'go]\label{def:T-QK}
Empiryczna relacja Koide'go \cite{Koide:1983qe} dla mas naladowanych
lepton\'ow $m_e, m_\mu, m_\tau$:
\begin{equation}\label{eq:T-QK}
  \boxed{Q_K \;\equiv\;
    \frac{\bigl(\sqrt{m_e} + \sqrt{m_\mu} + \sqrt{m_\tau}\bigr)^2}%
         {m_e + m_\mu + m_\tau}
    \;=\; \frac{3}{2}.}
\end{equation}
Relacja ta jest weryfikowana eksperymentalnie z~dok\l{}adno\'sci\k{a}
$|\Delta Q_K| < 4 \times 10^{-5}$ \cite{PDG2024}.
\end{definition}

\begin{remark}[Status formu\l{}y Koide'go w~TGP]\label{rem:T-koide-status}
W~TGP formu\l{}a~\eqref{eq:T-QK} \textbf{nie jest za\l{}o\.zeniem}:
jest traktowana jako \emph{empiryczny warunek selekcji}
nak\l{}adany na wyniki Twierdzenia~\ref{thm:J2-FP}.
Jej dynamiczne uzasadnienie w~ramach TGP
(np.\ symetria w~przestrzeni $g_0$-parametr\'ow solitonu)
pozostaje problemem otwartym T-OP2
(sekcja~\ref{app:T-open}).
\end{remark}

\subsection{Algebraiczne wyznaczenie $r_{31}$ z~$Q_K = 3/2$}%

\begin{theorem}[Wyznaczenie $r_{31}$ z~relacji Koide'go]%
\label{thm:T-r31}
Niech $r_{21} = m_\mu/m_e$ b\k{e}dzie dane.
Przy za\l{}o\.zeniu $Q_K = 3/2$ (def.~\ref{def:T-QK})
stosunek $r_{31} = m_\tau/m_e$ spe\l{}nia r\'ownanie kwadratowe
(wzgl\k{e}dem $\sqrt{r_{31}}$) o~\textbf{jednoznacznym} rozwi\k{a}zaniu
fizycznym:
\begin{equation}\label{eq:T-r31-formula}
  \boxed{\sqrt{r_{31}} \;=\; 2\bigl(1 + \sqrt{r_{21}}\bigr)
    \;+\; \sqrt{3\bigl(1 + 4\sqrt{r_{21}} + r_{21}\bigr)}.}
\end{equation}
Dla $r_{21} = 206{,}768$ (PDG):
\begin{equation}\label{eq:T-r31-num}
  \sqrt{r_{31}}
    = 2(1 + 14{,}380) + \sqrt{3 \times 265{,}288}
    = 30{,}759 + 28{,}211
    = 58{,}970,
  \qquad
  r_{31}^K = 3477{,}5.
\end{equation}
Odchylenie od PDG ($r_{31}^{\rm PDG} = 3477{,}18$):
$\delta r_{31} = 0{,}009\%$.

\textbf{Dow\'od}.
Podstawiamy $x = \sqrt{r_{21}}$, $y = \sqrt{r_{31}}$ do~\eqref{eq:T-QK}
po podzieleniu przez $m_e$:
\begin{equation}\label{eq:T-QK-norm}
  Q_K = \frac{(1 + x + y)^2}{1 + x^2 + y^2} = \frac{3}{2}.
\end{equation}
Przy $Q_K = 3/2$ r\'ownanie~\eqref{eq:T-QK-norm} przyjmuje postac:
\begin{equation}\label{eq:T-quad-raw}
  2(1 + x + y)^2 = 3(1 + x^2 + y^2).
\end{equation}
Rozwijaj\k{a}c i~porz\k{a}dkuj\k{a}c wzgl\k{e}dem $y$:
\begin{equation}\label{eq:T-quad-y}
  y^2 - 2(2x+2)\,y + (x^2 - 4x + 1) = 0,
\end{equation}
sk\k{a}d (wz\'or kwadratowy; wyznacznik
$\Delta_y = 4(2x+2)^2 - 4(x^2-4x+1) = 48(x^2+4x+1) > 0$):
\begin{equation}\label{eq:T-quad-sol}
  y = (2x+2) \pm \sqrt{3(x^2 + 4x + 1)}.
\end{equation}
Znak $+$ daje $y > 0$ dla $x \gg 1$; znak $-$ daje
$y < x$ ($m_\tau < m_\mu$), co jest fizycznie wykluczone.
St\k{a}d jednoznaczne rozwi\k{a}zanie~\eqref{eq:T-r31-formula}.
\hfill$\square$
\end{theorem}

\begin{theorem}[$Q_K = 3/2 \Leftrightarrow \mathrm{CV}(\sqrt{m}) = 1
  \Leftrightarrow \theta = 45^\circ$]%
\label{thm:T-QK-CV}
Niech $m_1, m_2, m_3 > 0$ b\k{e}d\k{a} masami trzech generacji.
Zdefiniujmy:
\begin{align}
  Q_K &:= \frac{(\sqrt{m_1}+\sqrt{m_2}+\sqrt{m_3})^2}{m_1+m_2+m_3},
    \label{eq:T-QK-def2}\\
  \mathrm{CV} &:= \frac{\sigma(\sqrt{m_i})}{\mu(\sqrt{m_i})}
    \;=\; \frac{\sqrt{\tfrac{1}{3}\sum_i(\sqrt{m_i}-\bar\lambda)^2}}{\bar\lambda},
    \label{eq:T-CV-def}\\
  \theta &:= \angle\!\bigl((\sqrt{m_1},\sqrt{m_2},\sqrt{m_3}),\;
    \tfrac{1}{\sqrt3}(1,1,1)\bigr),
    \label{eq:T-angle-def}
\end{align}
gdzie $\bar\lambda = \tfrac{1}{3}\sum_i\sqrt{m_i}$.
Wtenczas nast\k{e}puj\k{a}ce trzy warunki s\k{a} r\'ownowa\.zne:
\begin{equation}\label{eq:T-triple}
  \boxed{Q_K = \tfrac{3}{2}
  \;\iff\;
  \mathrm{CV}(\sqrt{m_i}) = 1
  \;\iff\;
  \theta = 45^\circ.}
\end{equation}

\begin{proof}
\textbf{(1) $Q_K = N/(1+\mathrm{CV}^2)$.}
Niech $\lambda_i = \sqrt{m_i}$, $\bar\lambda = \tfrac{1}{N}\sum_i\lambda_i$,
$\sigma^2 = \tfrac{1}{N}\sum_i(\lambda_i-\bar\lambda)^2$.
Wtedy $S_1 := \sum_i\lambda_i = N\bar\lambda$ oraz
$S_2 := \sum_i m_i = \sum_i\lambda_i^2 = N(\bar\lambda^2+\sigma^2)
  = N\bar\lambda^2(1+\mathrm{CV}^2)$.
St\k{a}d:
\begin{equation}
  Q_K = \frac{S_1^2}{S_2}
  = \frac{N^2\bar\lambda^2}{N\bar\lambda^2(1+\mathrm{CV}^2)}
  = \frac{N}{1+\mathrm{CV}^2}.
\end{equation}
Dla $N = 3$: $Q_K = 3/2 \Leftrightarrow 1+\mathrm{CV}^2 = 2
  \Leftrightarrow \mathrm{CV} = 1$.

\textbf{(2) $\cos^2\theta = Q_K/N$.}
Niech $\mathbf{v} = (\lambda_1,\lambda_2,\lambda_3)$,
$\mathbf{d} = (1,1,1)/\sqrt{3}$.
W\'owczas:
\begin{equation}
  \cos\theta = \frac{\mathbf{v}\cdot\mathbf{d}}{|\mathbf{v}|}
  = \frac{S_1/\sqrt{3}}{\sqrt{S_2}}
  = \sqrt{\frac{S_1^2}{3S_2}}
  = \sqrt{\frac{Q_K}{3}}.
\end{equation}
Dla $N=3$: $\theta = 45^\circ \Leftrightarrow \cos^2\theta = 1/2
  \Leftrightarrow Q_K = 3/2$.
\end{proof}

\end{theorem}

\begin{corollary}[Formulacja TGP: $\mathrm{CV}(A^2)=1$]%
\label{cor:T-CV-Atail}
W~TGP masy leptonowe spe\l{}niaj\k{a} $m_i \propto A(g_0^{(i)})^4$
(def.~\ref{def:A-tail}), sk\k{a}d $\sqrt{m_i} \propto A(g_0^{(i)})^2$.
Formu\l{}a Koide'go $Q_K = 3/2$ jest r\'ownowa\.zna warunkowi:
\begin{equation}\label{eq:T-CV-TGP}
  \mathrm{CV}\!\bigl(A(g_0^{(n)})^2\bigr) \;=\; 1,
  \qquad
  \text{tj.}\quad
  \sigma\!\bigl(A^2\bigr) \;=\; \bar{A^2},
\end{equation}
czyli \textbf{odchylenie standardowe kwadrat\'ow amplitud ogonowych
r\'owna si\k{e} ich \'{s}redniej}.
Numerycznie ($A_e^2:A_\mu^2:A_\tau^2 = 1:14{,}38:58{,}97$):
$\sigma(A^2) = \bar{A^2} = 24{,}783$ (PASS, 8/8).

\begin{proposition}[$Q_K = 3/2$ z~symetrii $\mathbb{Z}_3$ --- parametryzacja Brannena]%
\label{prop:T-Brannen-Z3}
Niech amplitudy ogonowe trzech generacji spe\l{}niaj\k{a}:
\begin{equation}\label{eq:T-Brannen}
  \sqrt{m_k} \;=\; c\!\left(1 + \sqrt{2}\cos\!\left(\theta_K + \frac{2\pi k}{3}\right)\right),
  \qquad k = 0, 1, 2,
\end{equation}
dla pewnych sta\l{}ych $c > 0$ i~$\theta_K \in \mathbb{R}$ (k\k{a}t Koide'go).
Wtenczas $Q_K = 3/2$ dla \textbf{ka\.zdego} $\theta_K$.

\begin{proof}
Oznaczmy $\phi_k = \theta_K + 2\pi k/3$. Korzystamy z~dw\'och
identyczno\'sci trygonometrycznych dla $\phi_k$ r\'ownomiernie
roz\l{}o\.zonych co $2\pi/3$:
\begin{align}
  \sum_{k=0}^{2} \cos\phi_k &= 0, \label{eq:T-trig1}\\
  \sum_{k=0}^{2} \cos^2\!\phi_k &= \tfrac{3}{2}. \label{eq:T-trig2}
\end{align}
St\k{a}d:
\begin{align}
  S_1 &= \sum_k\sqrt{m_k} = c\sum_k\!\left(1+\sqrt{2}\cos\phi_k\right)
        = 3c \;+\; c\sqrt{2}\underbrace{\sum_k\cos\phi_k}_{=\,0} = 3c, \\
  S_2 &= \sum_k m_k = c^2\sum_k\!\left(1+\sqrt{2}\cos\phi_k\right)^2 \notag\\
       &= c^2\!\left(3 + 2\sqrt{2}\underbrace{\sum_k\cos\phi_k}_{=\,0}
          + 2\underbrace{\sum_k\cos^2\!\phi_k}_{=\,3/2}\right) = 6c^2, \\
  Q_K &= \frac{S_1^2}{S_2} = \frac{9c^2}{6c^2} = \frac{3}{2}. \qquad \square
\end{align}
\end{proof}

wyra\.zaj\k{a} \textbf{symetri\k{e} $\mathbb{Z}_3$} (r\'ownomierne
rozmieszczenie faz co $2\pi/3$), nie za\'s specyficzne warto\'sci
$\phi_k$. Dlatego $Q_K = 3/2$ jest \emph{konsekwencj\k{a} symetrii

\begin{theorem}[$Q_K = 3/2$ z~$N_{\mathrm{gen}} = 3$
  i~parametryzacji Brannena]%
\label{thm:T-QK-from-Ngen}
Niech $N$ b\k{e}dzie liczb\k{a} generacji fermionowych
i~niech masy $m_k$ ($k = 1,\ldots,N$) b\k{e}d\k{a}
zapisane w~parametryzacji Brannena
(prop.~\ref{prop:T3-qkNr}):
\[
  \sqrt{m_k} = M\bigl(1 + r\cos(\theta + 2\pi k/N)\bigr),
  \qquad k = 0, 1, \ldots, N-1.
\]
Wtedy:
\begin{enumerate}[nosep]
  \item $Q_K^{(N)}(r) = N/(1 + r^2/2)$ (niezale\.znie od $\theta$ i~$M$).

  \item \textbf{Hipoteza (ekwipartycja amplitud):}
    Je\.zeli $N$ niezale\.znych mod\'ow solitonowych dzieli
    $N-1$ stopni swobody w~przestrzeni $\sqrt{m}$
    (simpleks $N-1$ wymiarowy),
    to parametr rozrzutu Brannena przyjmuje warto\'s\'c
    \begin{equation}\label{eq:T-r-Brannen-N}
      r = \sqrt{N - 1}.
    \end{equation}

  \item Z~(1) i~(2):
    \begin{equation}\label{eq:T-QK-general-N}
      Q_K^{(N)}\!\bigl(\sqrt{N-1}\bigr) = \frac{2N}{N+1}.
    \end{equation}

  \item \textbf{Unikalno\'s\'c $N = 3$ dla $Q_K = 3/2$:}
    R\'ownanie $2N/(N+1) = 3/2$ ma jedyne
    rozwi\k{a}zanie $N = 3$. Jednocze\'snie $N = 3$
    jest jedynym wymiarem, w~kt\'orym
    $r = \sqrt{N-1}$ daje $\mathrm{CV}(\sqrt{m}) = 1$
    (poniewa\.z $\mathrm{CV}^2 = r^2/2 = (N-1)/2$,
    a~$\mathrm{CV} = 1$ wymaga $N = 3$).
\end{enumerate}
\end{theorem}

\begin{proof}
Punkt (1): dowodziony w~prop.~\ref{prop:T3-qkNr}.

$r^2 = N-1$ do (1):
$Q_K = N/(1 + (N-1)/2) = N/((N+1)/2) = 2N/(N+1)$.
Punkt (4): $2N/(N+1) = 3/2 \Rightarrow 4N = 3(N+1)
\Rightarrow N = 3$.
Dla $\mathrm{CV}$: $\mathrm{CV}^2 = r^2/2 = (N-1)/2$;
$\mathrm{CV} = 1$ wymaga $(N-1)/2 = 1$, tj.~$N = 3$.
\end{proof}

\begin{proposition}[Informacyjno-teoretyczne wzmocnienie hipotezy ekwipartycji]%
\label{prop:T-equipartition-info}
Niech $\sqrt{m_k} = M(1 + r\cos(\theta + 2\pi k/N))$, $k = 0,\ldots,N{-}1$,
z~wi\k{e}zem $\sum m_k = M_{\rm total}$ (ustalona masa ca\l{}kowita sektora).

z~estymacj\k{a} $M$ (skali masowej) przy ustalonym kszta\l{}cie
($\theta$, $r$, $N$ ustalone):
\begin{equation}\label{eq:T-Fisher}
  \mathcal{I}_F(r, N)
  = \sum_{k=0}^{N-1}
    \frac{1}{m_k}\!\left(\frac{\partial m_k}{\partial M}\right)^{\!2}
  = \sum_{k=0}^{N-1}
    \frac{[1 + r\cos(\theta + 2\pi k/N)]^2}{M}.
\end{equation}

$\sum_{k} \cos^2(\theta + 2\pi k/N) = N/2$ dla $N \ge 3$:
\begin{equation}
  \mathcal{I}_F = \frac{1}{M}\!\left(N + \tfrac{N}{2}\,r^2\right)
  = \frac{N}{M}\!\left(1 + \tfrac{r^2}{2}\right).
\end{equation}

\textbf{rosn\k{a}c\k{a}} w~$r$ --- nie daje minimum.

specyficznego dla TGP: ca\l{}kowita amplituda ogona soliton\'ow

    Spearmana $\rho_S = 0{,}33$ ($p = 0{,}054$):
    spirala jest \textbf{s\l{}abo sprz\k{e}\.zona}, ale
    nie pe\l{}ni okr\k{a}g\l{}o-symetryczna.
  \item $Q_K$ maleje \textbf{monotonicznie} od $2{,}12$ do $1{,}32$
    w~miar\k{e} wzrostu $g_{0,\tau}$
    --- brak ekstremum, brak mechanizmu samo-selekcji.
\end{enumerate}

\medskip\noindent
\textbf{Konkluzja:}
Ramy $\chi^2(2)$ wyja\'sniaj\k{a}, \emph{dlaczego}
Koide dzia\l{}a na poziomie masy
(inwariancja $A^2 = a^2 + b^2$),
ale \textbf{nie derywuj\k{a}} $\mathrm{CV} = 1$.
Po 21 \'scie\.zkach analizy, $Q_K = 3/2$ pozostaje
\textbf{parametrem wej\'sciowym} (selekcja $g_{0,\tau}$).
Otwarte pytanie: co fizycznie \textbf{selekcjonuje}
$g_{0,\tau} = 1{,}570$ w~korytarzu kolapsu $[1{,}404;\; 1{,}600)$?

Skrypty: \texttt{koide\_chi2\_closure\_v47b.py},
\texttt{koide\_chi2\_closure\_fast.py}.
\end{remark}

\begin{proposition}[Pitagorejski mechanizm selekcji $Q_K = 3/2$]%
\label{prop:T-pythagorean-selection}
W~kanonicznym TGP ($K(g)=g^4$, $d=3$) \l{}\k{a}cznik dedukcyjny
od amplitud ogonowych do relacji Koide'go ma nast\k{e}puj\k{a}c\k{a}
struktur\k{e}:

\medskip\noindent
\textbf{Krok~1 (ODE $\to$ dwie sk\l{}adowe):}
Linearyzacja kanonicznego ODE wok\'o\l{} pr\'o\.zni ($g \to 1$)
daje r\'ownanie sferyczne Bessela z~$\omega = 1$
(prop.~\ref{prop:T-oscillatory-tail}).
Og\'olne rozwi\k{a}zanie w~$d=3$:
\begin{equation}\label{eq:T-tail-ab}
  h(r) \;:=\; \bigl(g(r)-1\bigr)\cdot r
  \;\xrightarrow{r\to\infty}\;
  a\,\cos r \;+\; b\,\sin r,
\end{equation}
gdzie $a = a(g_0)$, $b = b(g_0)$ s\k{a} \textbf{deterministycznymi}
funkcjami warunku pocz\k{a}tkowego, wyznaczonymi przez ca\l{}kowanie
pe\l{}nego (nieliniowego) ODE od $r = 0$ do ogona.

\medskip\noindent
\textbf{Krok~2 (suma pitagorejska):}
Amplituda ogonowa spe\l{}nia
\begin{equation}\label{eq:T-A2-pythagoras}
  A^2(g_0) \;=\; a(g_0)^2 \;+\; b(g_0)^2,
\end{equation}
co jest \textbf{inwariancj\k{a} obrotow\k{a}} w~p\l{}aszczy\'znie
$(a,b)$: $A^2$ zale\.zy od \textbf{promienia},
nie od fazy $\theta(g_0) = \arctan(b/a)$.

\medskip\noindent
\textbf{Krok~3 (mapa spiralna):}
Ca\l{}kowanie ODE dla $g_0 \in (1, g_{0,\mathrm{crit}})$
daje map\k{e} $g_0 \mapsto (a(g_0), b(g_0))$,
kt\'ora kre\'sli \textbf{spiral\k{e}} w~p\l{}aszczy\'znie $(a,b)$:
amplituda $A(g_0)$ ro\'snie monotonicznie,
faza $\theta(g_0)$ obraca si\k{e} z~pr\k{e}dko\'sci\k{a}
rosn\k{a}c\k{a} ku progowi kolapsu
($d\theta/dg_0$: od $\sim\!100^\circ$/dg$_0$ blisko pr\'o\.zni
do $>2000^\circ$/dg$_0$ blisko $g_{0,\mathrm{crit}} = 8/5$).

\medskip\noindent
\textbf{Krok~4 (selekcja $g_{0,\tau}$):}
Regu\l{}a $\varphi$-FP wyznacza $g_0^e$ i~$g_0^\mu = \varphi\,g_0^e$.
Warunek Koide'go $Q_K = 3/2$ (r\'ownowa\.znie
$\mathrm{CV}(A^2) = 1$, tw.~\ref{thm:T-QK-CV},
wn.~\ref{cor:T-CV-Atail}) \textbf{jednoznacznie} wyznacza
\begin{equation}\label{eq:T-g0tau-selection}
  g_{0,\tau} \;=\; \bigl(A^2\bigr)^{-1}\!\!
  \bigl(\text{jedyne $x$ takie, \.ze }
  \mathrm{CV}(A_e^2, A_\mu^2, x) = 1\bigr)
  \;\in\; \bigl[g_0^\mu,\; g_{0,\mathrm{crit}}\bigr).
\end{equation}
Monotoniczno\'s\'c $A^2(g_0)$ nad pr\'o\.zni\k{a}
gwarantuje \textbf{istnienie i~jednoznaczno\'s\'c} rozwi\k{a}zania.
Numerycznie: $g_{0,\tau} = 1{,}5696$ (tj.\ $1{,}9\%$ poni\.zej
$g_{0,\mathrm{crit}} = 1{,}6$).

\medskip\noindent
\textbf{Pe\l{}ny \l{}\k{a}cuch selekcji:}
\begin{equation}\label{eq:T-selection-chain}
  \boxed{
  g_0
  \;\xrightarrow{\;\text{ODE}\;}
  \bigl(a(g_0),\; b(g_0)\bigr)
  \;\xrightarrow{\;\text{Pitagoras}\;}
  A^2 = a^2 + b^2
  \;\xrightarrow{\;Q_K = 3/2\;}
  g_{0,\tau} = 1{,}5696.}
\end{equation}

\medskip\noindent
\textbf{Co ten mechanizm wyja\'snia:}
\begin{enumerate}[nosep]
  \item \textbf{Dlaczego masa, nie amplituda?}
    Poniewa\.z $A^2 = a^2+b^2$ u\'srednia po fazie spiralnej ---
    inwariancja obrotowa \l{}\k{a}czy Koide z~geometri\k{a}
    pitagorejsk\k{a} w~przestrzeni $(a,b)$, kt\'ora istnieje
    tylko dla $d \geq 3$ (oscylacyjne ogony).
  \item \textbf{Dlaczego $Q_K$ mo\.ze mie\'c r\'o\.zne warto\'sci?}
    Przesuni\k{e}cie $g_{0,\tau}$ w~korytarzu kolapsu
    daje $Q_K \in [1{,}27;\; 2{,}13]$. Warto\'s\'c $3/2$ jest
    \textbf{selekcjonowana}, nie wymuszona algebraicznie.
  \item \textbf{Co pozostaje otwarte?}
    Fizyczny mechanizm, kt\'ory \textbf{preferuje}
    $\mathrm{CV}(A^2) = 1$ (tj.\ $\sigma = \bar\mu$) spo\'sr\'od
    continuum dopuszczalnych warto\'sci.
    Jest to jedyne pozosta\l{}e otwarte pytanie
    programu Koide w~TGP.
\end{enumerate}
\end{proposition}

\begin{remark}[Dlaczego $\mathrm{CV} = 1$? --- \'scie\.zka 22 (v47b)]%
\label{rem:T-cv1-origin}
Prop.~\ref{prop:T-pythagorean-selection} redukuje ca\l{}y program Koide
do jednego pytania: \emph{dlaczego} $\mathrm{CV}(A^2) = 1$?
Zbadano dziewi\k{e}\'c niezale\.znych podej\'s\'c:

\begin{enumerate}[nosep]
  \item \textbf{Samopodoba A$^2$}: stosunek $A^2(\varphi g_0)/A^2(g_0)$
    zmienia si\k{e} od $\sim 0$ do $\sim 165$; mapa $A^2$ \textbf{nie jest}
    samopodobna wzgl\k{e}dem $\varphi$-skalowania.
  \item \textbf{Zasada wariacyjna}: \.zaden z~testowanych funkcjona\l{}\'ow
    (energia, entropia, AM/GM, iloczyn) nie ma ekstremum przy $Q_K = 3/2$
    --- wszystkie monotoniczne w~$g_{0,\tau}$.
  \item \textbf{Przybli\.zenie analityczne}:
    $A^2(g_0) \approx C\,(g_0{-}1)^{2{,}08}/(g_{0,\mathrm{crit}}{-}g_0)^{0{,}37}$
    z~dok\l{}adno\'sci\k{a} $0{,}4\%$.
    Nie implikuje algebraicznie $\mathrm{CV} = 1$.
  \item \textbf{Regu\l{}y sumowe}: szybko\'sci wzrostu $d(\ln A^2)/dg_0$
    przy trzech $g_0$ nie spe\l{}niaj\k{a} \.zadnej prostej regu\l{}y.
  \item \textbf{Dyskretne RG}: krok $g_0 \to \varphi g_0$
    \l{}amie si\k{e} przy $k = 2$ (kolaps).
  \item \textbf{Odleg\l{}o\'s\'c od kolapsu}: $\delta_\mu/\delta_\tau
    = 6{,}45$, brak dopasowania do znanych sta\l{}ych.
  \item \textbf{Entropia}: maksimum Shannona pada na $g_{0,\tau} = 1{,}42$,
    nie na Koide ($1{,}57$).
\end{enumerate}

\medskip\noindent
\textbf{Kluczowe odkrycie --- zbie\.zno\'s\'c dw\'och argument\'ow:}

Hipoteza ekwipartycji (tw.~\ref{thm:T-QK-from-Ngen})
z~formu\l{}\k{a} Brannena daje $b = \sqrt{N{-}1}$, sk\k{a}d
$\mathrm{CV}^2 = b^2/2 = (N{-}1)/2$.
Dla $N = 3$: $\mathrm{CV}^2 = 1$, $Q_K = 3/2$.
\textbf{To jest ta sama ,,dw\'ojka''} co w~hipotezie $\chi^2(2)$
(rem.~\ref{rem:T-substrate-chi2}):
\[
  \underbrace{2\;\text{sk\l{}adowe ogona}\;(a,b)}_{\chi^2(2)}
  \;=\;
  \underbrace{N - 1 = 2\;\text{niezale\.zne mody}}_{\text{Brannen}}.
\]
Oba argumenty prowadz\k{a} do $\mathrm{CV} = 1$ z~tej samej
przyczyny: w~$d = 3$ ogon ma \textbf{dwa} stopnie swobody,
a~$N = 3$ generacje z~jednym wi\k{a}zem maj\k{a} $N{-}1 = 2$
niezale\.zne mody. Postawiona teza: oba ,,2'' to ta sama
wielko\'s\'c --- \textbf{wymierno\'s\'c faz w~oscylacyjnym ogonie
w~$d = 3$ nadaje dwa mody, kt\'ore poprzez sum\k{e}
pitagorejsk\k{a} wymuszaj\k{a} $Q_K = 3/2$}.

\medskip\noindent
\textbf{Otwarta luka:}
Formalna implikacja
$d = 3 \;\wedge\; N_{\mathrm{gen}} = 3 \;\Longrightarrow\;
\mathrm{CV}(A^2) = 1$ wymaga dowodu,
\.ze $N{-}1$ mod\'ow generacyjnych wnosi
\textbf{jednakowy wk\l{}ad} do wariancji $A^2$.
Hipoteza ekwipartycji jest fizycznie motywowana
(rem.~\ref{rem:T-equipartition-status}),
ale nie dowiedziona z~pierwszych zasad ODE.

Skrypt: \texttt{koide\_cv1\_origin\_v47b.py}.
\end{remark}

\begin{remark}[Hipoteza ekwipartycji: test ko\'ncowy (v47b)]%
\label{rem:T-equipartition-test}
\'Scie\.zka~23 bada, czy hipotez\k{e} ekwipartycji mo\.zna
wyprowadzi\'c z~w\l{}asno\'sci ODE solitonowego.

\medskip\noindent
\textbf{Kluczowe wyniki:}
\begin{enumerate}[nosep]
  \item \textbf{Proporcjonalno\'s\'c $A^2 \propto I_{\mathrm{kin}}$:}
    Ca\l{}ka kinetyczna $I_{\mathrm{kin}} = \int g^4 g'^2 r^2\,dr$ jest
    \textbf{liniowo} proporcjonalna do $A^2$ z~wyk\l{}adnikiem
    $\alpha = 1{,}000 \pm 0{,}002$ (sta\l{}y dla wszystkich trzech
    generacji).
    Zatem $\mathrm{CV}(A^2) = 1$ jest r\'ownowa\.zne
    $\mathrm{CV}(I_{\mathrm{kin}}) = 1$.
  \item \textbf{Uniwersalno\'s\'c}: Dla \textbf{ka\.zdego} $g_0^e$
    z~zakresu $[0{,}50;\; 0{,}92]$ istnieje $g_{0,\tau}$ daj\k{a}ce
    $Q_K = 3/2$. Warto\'s\'c $Q_K = 3/2$ nie jest specyficzna
    dla fizycznego $g_0^e$ --- jest \textbf{zawsze osi\k{a}galna}.
  \item \textbf{Brak relacji funkcyjnej}: stosunek
    $g_{0,\tau}/g_{0,\mu}$ zmienia si\k{e} od $1{,}07$ do $1{,}31$;
    $\theta_{\mathrm{Brannen}}$ od $110^\circ$ do $135^\circ$;
    \.zaden z~tych parametr\'ow nie jest sta\l{}\k{a} uniwersaln\k{a}.
  \item \textbf{Skalowanie bliskie kolapsowi}:
    $\delta_\tau \approx C\,\delta_\mu^{1{,}86}$
    (najlepsze dopasowanie; $1{,}86$ bliskie $2$
    z~b\l{}\k{e}dem $7\%$).
  \item \textbf{Energia monotonowa}: suma energii soliton\'ow
    $E_e + E_\mu + E_\tau(g_{0,\tau})$ jest monotonicznie
    malej\k{a}ca --- \.zaden funkcjona\l{} energetyczny nie
    selekcjonuje Koide.
\end{enumerate}

\textbf{Status ko\'ncowy problemu $Q_K = 3/2$ w~TGP:}\\

\end{itemize}

\end{remark}
